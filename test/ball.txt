import threading

class Ball:
	def __init(self):
		self.angle = None
		self.distance = None
		self.color = None
		self.time = None

	def setDistance(self, dist):
		self.distance = dist

	def getDistance(self):
		return self.distance

	def setAngle(self, ang):
		self.angle = ang

	def getAngle(self):
		return self.angle

	def setColor(self, c):
		self.color = c

	def getColor(self):
		return self.color

	def getTime(self):
		return self.time

	def setTime(self, t):
		self.time = t
